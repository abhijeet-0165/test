#!/bin/bash
a=$1
b=$2

fi

Q3. Write a shell script that defines a function called 'find_stats' which accepts an array of integers as arguments. The function should calculate and display the count of elements, the sum of all elements, the minimum value, and the maximum value. Call the function with the values: 45 12 78 3 56 91 24.
Solution:
#!/bin/bash
find_stats() {
    local arr=("$@")
    local count=${arr[@]}
    local sum=0
    local min=${arr[0]}
    local max=${arr[0]}
    for val in "${arr[@]}"; do
        sum=$((sum + val))
        if [ $val -lt $min ]; then
            min=$val
        fi
        if [ $val -gt $max ]; then
            max=$val
        fi
    done
    echo "Count: $count"
    echo "Sum: $sum"
    echo "Min: $min"
    echo "Max: $max"
}
find_stats 45 12 78 3 56 91 24
Q4. Write a shell script that reads a directory path as a command-line argument and performs the following:
(i) Check if the directory exists. If not, display an error message and exit.
(ii) Count and display the total number of files and directories inside the given path.
(iii) For each item, use a 'for' loop and 'case' or 'if' to classify it as a 'File' or 'Directory' and print its name with the classification.
Solution:
#!/bin/bash
dirpath=$1
if [ ! -d "$dirpath" ]; then
    echo "Error: Directory does not exist"
    exit 1
fi
file_count=0
dir_count=0
for item in "$dirpath"/*; do
    if [ -f "$item" ]; then
        file_count=$((file_count + 1))
        echo "File: $(basename $item)"
    elif [ -d "$item" ]; then
        dir_count=$((dir_count + 1))
        echo "Directory: $(basename $item)"
    fi
done
echo "Total Files: $file_count"
echo "Total Directories: $dir_count"

Q3. Write a shell script that accepts a filename and an operation as command-line arguments. The operation can be 'lines', 'words', or 'chars'. Using a case statement, the script should display the count of lines, words, or characters in the file respectively. If an invalid operation is given, display an error message.
Solution:
#!/bin/bash
filename=$1
operation=$2
if [ ! -f "$filename" ]; then
    echo "Error: File not found"
    exit 1
fi
case $operation in
    lines)
        count=$(wc -l < "$filename")
        echo "Lines: $count"
        ;;
    words)
        count=$(wc -w < "$filename")
        echo "Words: $count"
        ;;
    chars)
        count=$(wc -c < "$filename")
        echo "Characters: $count"
        ;;
    *)
        echo "Error: Invalid operation. Use lines, words, or chars"
        exit 1
        ;;
esac
Q4. Write a shell script that:
(i) Defines an array with elements: ubuntu debian fedora centos arch
(ii) Defines a function 'search_distro' that accepts a search term and checks if it exists in the array.
(iii) If found, print the element and its index. If not found, print a 'not found' message.
(iv) Call the function with 'fedora' and then with 'mint'.
Solution:
#!/bin/bash
distros=(ubuntu debian fedora centos arch)

search_distro() {
    local search=$1
    local found=0
    for i in "${!distros[@]}"; do
        if [ "${distros[$i]}" == "$search" ]; then
            echo "'$search' found at index $i"
            found=1
            break
        fi
    done
    if [ $found -eq 0 ]; then
        echo "'$search' not found in the array"
    fi
}
search_distro "fedora"
search_distro "mint"

Q3. Write a shell script that uses an 'until' loop to read user input repeatedly. For each input string:
(i) Display the string in uppercase using parameter expansion.
(ii) Display the length of the string.
(iii) Display the string reversed.
The loop should stop when the user enters 'quit'.
Solution:
#!/bin/bash
while IFS= read -r line; do
    if [ "$line" == "quit" ]; then
        echo "Exiting..."
        break
    fi
    echo "Uppercase: ${line^^}"
    echo "Length: ${line}"
    reversed=""
    for (( i=${line}-1; i>=0; i-- )); do
        reversed="$reversed${line}"
    done
    echo "Reversed: $reversed"
    echo "---"
done < /tmp/input.txt
Q4. Write a shell script that accepts a service name as a command-line argument and performs the following:
(i) Check if the service is currently running using systemctl.
(ii) If running, display 'Service is active' and show its PID.
(iii) If not running, display 'Service is inactive' and attempt to start it.
(iv) Display the enabled/disabled status of the service.
Solution:
#!/bin/bash
service=$1
if systemctl is-active --quiet "$service"; then
    echo "Service '$service' is active"
    pid=$(systemctl show -p MainPID "$service" | cut -d= -f2)
    echo "PID: $pid"
else
    echo "Service '$service' is inactive"
    echo "Attempting to start..."
    systemctl start "$service" 2>/dev/null
fi
status=$(systemctl is-enabled "$service" 2>/dev/null)
echo "Boot status: $status"
